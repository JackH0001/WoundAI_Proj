```python
class ImagePreprocessor:
    def __init__(self):
        self.ruler_calibrator = RulerCalibrationSystem()
        self.quality_assessor = ImageQualityAssessor()
    
    def process(self, image):
        # 1. 品質檢查
        quality_score = self.quality_assessor.assess(image)
        if quality_score < 0.7:
            return None, "影像品質不符合標準"
        
        # 2. 標尺校正
        calibrated_image, scale_factor = self.ruler_calibrator.calibrate(image)
        
        # 3. 去雜訊和正規化
        processed_image = self.denoise_and_normalize(calibrated_image)
        
        return processed_image, scale_factor
```

#### 3. **自動化指標計算模組**
```python
class AutomatedScoring:
    def calculate_bjwat_scores(self, image, masks, scale_factor):
        scores = {}
        
        # 1. 傷口大小 (可完全自動化)
        wound_area_cm2 = self.calculate_area(masks['wound_boundary'], scale_factor)
        scores['bjwat_size'] = self.area_to_size_grade(wound_area_cm2)
        
        # 2. 壞死量 (可完全自動化)
        necrotic_ratio = self.calculate_area_ratio(
            masks['necrotic_tissue'], masks['wound_boundary']
        )
        scores['bjwat_necrotic_amount'] = self.ratio_to_amount_grade(necrotic_ratio)
        
        # 3. 組織顏色 (可完全自動化)
        color_features = self.extract_color_features(image, masks['granulation'])
        scores['bjwat_tissue_color'] = self.color_to_grade(color_features)
        
        # 4. 邊緣特徵 (可完全自動化)
        edge_features = self.analyze_wound_edges(image, masks['wound_boundary'])
        scores['bjwat_edges'] = self.edge_features_to_grade(edge_features)
        
        # 5. 肉芽組織 (可完全自動化)
        granulation_ratio = self.calculate_area_ratio(
            masks['granulation'], masks['wound_boundary']
        )
        scores['bjwat_granulation'] = self.granulation_to_grade(granulation_ratio)
        
        return scores
```

```python
class WoundSegmentationModel(nn.Module):
    def __init__(self, num_classes=6):
        super().__init__()
        # 類別: 背景、傷口邊界、壞死組織、肉芽組織、上皮化區、分泌物
```

```python
class UncertaintyAwareScoring:
    def calculate_with_uncertainty(self, features):
        # 蒙地卡羅Dropout
        predictions = []
        for _ in range(50):
            pred = self.model(features, training=True)
            predictions.append(pred)
        
        mean_pred = np.mean(predictions, axis=0)
        uncertainty = np.std(predictions, axis=0)
        
        return mean_pred, uncertainty
```

```python
class AutomatedUploadManager:
    async def process_and_upload(self, image, patient_info):
        try:
            # 1. 本地自動分析
            analysis_result = await self.wound_analyzer.analyze(image)
            
            # 2. 置信度檢查
            if analysis_result.confidence < 0.8:
                # 低置信度 - 標記需要人工複審
                analysis_result.requires_manual_review = True
                self.queue_for_manual_review(analysis_result)
            
            # 3. 建構上傳資料
            upload_data = {
                "patient_id": patient_info.id,
                "doctor_id": self.current_doctor.id,
                "automated_scores": {
                    **analysis_result.bjwat_scores,
                    **analysis_result.revpwat_scores
                },
                "confidence_scores": analysis_result.confidence_by_metric,
                "requires_review": analysis_result.requires_manual_review,
                "processing_metadata": {
                    "model_version": self.model_version,
                    "processing_time": analysis_result.processing_time,
                    "device_info": self.device_info
                }
            }
            
            # 4. 上傳到雲端API
            response = await self.api_client.upload_annotation(
                annotation_data=json.dumps(upload_data),
                image=image
            )
            
            # 5. 處理回應
            if response.success:
                self.save_local_record(analysis_result, response.annotation_id)
                return UploadResult.success(response)
            else:
                return UploadResult.failure(response.error)
                
        except Exception as e:
            logger.error(f"自動上傳失敗: {e}")
            # 儲存到本地佇列，稍後重試
            self.queue_for_retry(image, patient_info)
            return UploadResult.queued()
```

```python
# 在雲端API新增自動化處理端點
@app.post("/upload/automated_annotation")
@limiter.limit("10/minute") 
async def upload_automated_annotation(
    request,
    background_tasks: BackgroundTasks,
    annotation_data: str = Form(...),
    image: UploadFile = File(...),
    confidence_threshold: float = Form(0.8)
):
    """接收行動端自動化分析結果"""
    try:
        # 解析自動化分析結果
        auto_data = json.loads(annotation_data)
        
        # 提取置信度資訊
        confidence_scores = auto_data.get("confidence_scores", {})
        overall_confidence = np.mean(list(confidence_scores.values()))
        
        # 決定處理流程
        if overall_confidence >= confidence_threshold:
            # 高置信度 - 直接接受
            processing_status = "accepted"
            next_action = "approved"
        else:
            # 低置信度 - 排入人工複審佇列
            processing_status = "pending_review"
            next_action = "manual_review_required"
            
            # 背景任務：通知專家複審
            background_tasks.add_task(
                notify_expert_review, 
                auto_data["doctor_id"], 
                annotation_data
            )
        
        # 回應結果
        return {
            "success": True,
            "annotation_id": f"auto_{int(time.time())}",
            "processing_status": processing_status,
            "next_action": next_action,
            "overall_confidence": round(overall_confidence, 3),
            "automated_scores": auto_data["automated_scores"],
            "requires_expert_review": overall_confidence < confidence_threshold,
            "estimated_review_time": "2-4小時" if processing_status == "pending_review" else "立即"
        }
        
    except Exception as e:
        logger.error(f"自動化標註上傳失敗: {e}")
        raise HTTPException(status_code=500, detail="自動化處理失敗")
```
