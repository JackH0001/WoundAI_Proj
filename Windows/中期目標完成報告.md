### 技術架構升級

#### 資料庫架構
```sql
-- 醫師資料表
CREATE TABLE doctors (
    id VARCHAR(50) PRIMARY KEY,
    doctor_id VARCHAR(20) UNIQUE NOT NULL,
    name VARCHAR(100) NOT NULL,
    hospital VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE,
    hashed_password VARCHAR(255) NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    is_verified BOOLEAN DEFAULT FALSE,
    role VARCHAR(20) DEFAULT 'doctor',
    permissions TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    last_login TIMESTAMP
);

-- 醫師會話表
CREATE TABLE doctor_sessions (
    id VARCHAR(50) PRIMARY KEY,
    doctor_id VARCHAR(20) NOT NULL,
    session_token VARCHAR(255) UNIQUE NOT NULL,
    ip_address VARCHAR(45),
    user_agent TEXT,
    device_info TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    expires_at TIMESTAMP NOT NULL,
    last_activity TIMESTAMP DEFAULT NOW(),
    is_active BOOLEAN DEFAULT TRUE
);

-- 傷口標註資料表
CREATE TABLE wound_annotations (
    id VARCHAR(50) PRIMARY KEY,
    annotation_id VARCHAR(50) UNIQUE NOT NULL,
    doctor_id VARCHAR(20) NOT NULL,
    image_uri VARCHAR(500),
    image_width INTEGER,
    image_height INTEGER,
    image_format VARCHAR(10),
    file_size_bytes INTEGER,
    bjwat_total_score INTEGER DEFAULT 0,
    revpwat_total_score INTEGER DEFAULT 0,
    quality_score FLOAT DEFAULT 0.0,
    annotation_quality VARCHAR(20) DEFAULT 'pending',
    additional_notes TEXT,
    segmentation_mask TEXT,
    bounding_box TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP
);
```

#### API 端點設計
```
POST /auth/login          # 醫師登入
POST /auth/register       # 醫師註冊 (管理員功能)
POST /auth/logout         # 醫師登出
GET  /auth/verify         # 令牌驗證

POST /upload/annotation   # 上傳標註資料
POST /upload/image        # 上傳影像檔案
GET  /upload/progress/{id} # 查詢上傳進度

GET  /health              # 健康檢查
GET  /                    # 根路徑
```
